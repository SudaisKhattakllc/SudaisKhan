interface Props {
  project: Project;
  delayClass: string;
}

export default function ProjectCard({ project, delayClass }: Props) {
  const artIndex = parseInt(project.number) - 1;

  return (
    <div
      className={clsx(
        styles.card,
        "glass-card card-lift reveal",
        delayClass,
        // Grid spans come from data so layout is data-driven
        project.colSpan,
        project.rowSpan
      )}
    >
      {/* Line art */}
      <div className={styles.art} aria-hidden="true">
        {ARTS[artIndex % ARTS.length]}
      </div>

      {/* Card number */}
      <span className={styles.num}>{project.number}</span>

      {/* Tags */}
      <div className={styles.tags}>
        {project.tags.map((tag) => (
          <span key={tag} className={styles.tag}>{tag}</span>
        ))}
      </div>

      {/* Footer label */}
      <div className={styles.label}>
        <h3 className={styles.title}>{project.title}</h3>
        <p className={styles.tech}>{project.tech}</p>
      </div>

      {/* Glass overlay */}
      <div className={styles.glassOverlay} aria-hidden="true" />
    </div>
  );
}

/* ── Small info row ───────────────────────────────────────── */
function InfoItem({ icon, label, value, href }: {
  icon: "email" | "location" | "clock";
  label: string;
  value: string;
  href?: string;
}) {
  const ICONS = {
    email: <path d="M1 4l7 5 7-5M1 2h14a1 1 0 011 1v10a1 1 0 01-1 1H1a1 1 0 01-1-1V3a1 1 0 011-1z"/>,
    location: <><path d="M8 1C5.24 1 3 3.24 3 6c0 4.25 5 9 5 9s5-4.75 5-9c0-2.76-2.24-5-5-5z"/><circle cx="8" cy="6" r="1.5"/></>,
    clock: <><circle cx="8" cy="8" r="7"/><path d="M8 4v4l3 2"/></>,
  };

  return (
    <div style={{display:"flex",gap:"16px",alignItems:"center",marginBottom:"24px"}}>
      <div style={{
        width:36,height:36,background:"var(--white)",borderRadius:"50%",
        display:"flex",alignItems:"center",justifyContent:"center",flexShrink:0
      }}>
        <svg width="16" height="16" viewBox="0 0 16 16" fill="none" stroke="black" strokeWidth="1">
          {ICONS[icon]}
        </svg>
      </div>
      <div>
        <div style={{fontFamily:"var(--font-mono)",fontSize:10,letterSpacing:"0.18em",textTransform:"uppercase",color:"var(--gray-600)",marginBottom:4}}>
          {label}
        </div>
        {href
          ? <a href={href} style={{fontFamily:"var(--font-mono)",fontSize:13,color:"var(--white)",textDecoration:"none"}}>{value}</a>
          : <span style={{fontFamily:"var(--font-mono)",fontSize:13,color:"var(--white)"}}>{value}</span>
        }
      </div>
    </div>
  );
}
